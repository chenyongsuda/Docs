

题目描述：
大家都知道斐波那契数列，现在要求输入一个整数n，请你输出斐波那契数列的第n项。斐波那契数列的定义如下：

输入：
输入可能包含多个测试样例，对于每个测试案例，
输入包括一个整数n(1<=n<=70)。
输出：
对应每个测试案例，
输出第n项斐波那契数列的值。
样例输入：
3
样例输出：
2


#include<stdio.h>
#include<string.h>


int main()
{
	int n;
	//Build Linked list
	while(scanf("%d",&n) != EOF) {
		int fir = 0, sec = 1;
		if(0 == n){
			printf("0\n");
			continue;
		}
		if(1 == n){
			printf("1\n");
			continue;
		}
		for(int i = 2; i <= n; i++){
			int tmp = sec;
			sec = fir + sec;
			fir = tmp;
		}
		printf("%d\n",sec);
	}
	return 0;
}
