题目1368：二叉树中和为某一值的路径

题目描述：
输入一颗二叉树和一个整数，打印出二叉树中结点值的和为输入整数的所有路径。路径定义为从树的根结点开始往下一直到叶结点所经过的结点形成一条路径。
输入：
每个测试案例包括n+1行：
第一行为2个整数n，k(1<=n<=10000)，n表示结点的个数，k表示要求的路径和，结点编号从1到n。                                                                                                       
接下来有n行。这n行中每行为3个整数vi,leftnode,rightnode，vi表示第i个结点的值，leftnode表示第i个结点的左孩子结点编号，rightnode表示第i个结点的右孩子结点编号，若无结点值为-1。编号为1的结点为根结点。
输出：
对应每个测试案例，先输出“result:”占一行，接下来按字典顺序输出满足条件的所有路径，这些路径由结点编号组成，输出格式参照输出样例。
样例输入：
5 22
10 2 3
5 4 5
12 -1 -1
4 -1 -1
7 -1 -1
1 5
1 -1 -1
样例输出：
result:
A path is found: 1 2 5
A path is found: 1 3
result:

WA代码 没有安装字典序输出

#include <stdio.h> 
#include <string.h>

#define MAX 10001

typedef struct Node{
	int index;
	int val;
	int left;
	int right;
}Node;

Node* a[MAX];
int end = -1;
int seq[MAX];

void DFS(Node* root,int k,int sum){
	if(NULL == root){
		return;
	}
	sum += root->val;
	seq[++end] = root->index;
	if(sum == k){
		printf("A path is found:");
		for(int i = 0; i <= end; i++){
			printf(" %d",seq[i]);
		}
		printf("\n");
	}
	
	if(-1 != root->left){
		DFS(a[root->left],k, sum);
	}
	if(-1 != root->right){
		DFS(a[root->right],k, sum);
	}
	sum -= root->val;
	end--;
}

int main()
{
	int n,k;
	while(scanf("%d%d",&n,&k) != EOF) {
		int val,le,ri;
		end = -1;
		Node* node;
		int sum = 0;
		for (int i = 0; i < n; i++) {
			scanf("%d%d%d",&val,&le,&ri);
			node = new Node;
			node->index = i + 1;
			node->val = val;
			node->left = le;
			node->right = ri;
			a[i + 1] = node;
		}
		printf("result:\n");
		DFS(a[1],k,sum);
	}
	return 0;
}

输出要字典序输出 第一次4个case过了一个 主要是没注意字典序输出

怎么字典序输出呢？

因为先是递归左边子树  后递归右子树  可以比较左右子树大小  要是左子树大就交换左右子树

一直不AC 原因是只要头节点到叶节点的路径 不是所有的路径  ~~~

AC代码

#include <stdio.h> 
#include <string.h>

#define MAX 10001

typedef struct Node{
	int index;
	int val;
	int left;
	int right;
}Node;

Node* a[MAX];
int end = -1;
int seq[MAX];

void DFS(Node* root,int k,int sum){
	if(NULL == root){
		return;
	}
	sum += root->val;
	seq[++end] = root->index;
	if(sum == k && (root->left == -1) && (root->right == -1)){
		printf("A path is found:");
		for(int i = 0; i <= end; i++){
			printf(" %d",seq[i]);
		}
		printf("\n");
	}
	
	if(-1 != root->left){
		DFS(a[root->left],k, sum);
	}
	if(-1 != root->right){
		DFS(a[root->right],k, sum);
	}
	sum -= root->val;
	end--;
}

void ChangeTree(Node* root){
	if(NULL == root){
		return ;
	}
	if(-1 != root->left && -1 != root->right){
		if(root->left > root->right){
			int tmp = root->left;
			root->left = root->right;
			root->right = tmp;
		}
	}
	if(-1 != root->left)
		ChangeTree(a[root->left]);
	if(-1 != root->right)
		ChangeTree(a[root->right]);
} 

int main()
{
	int n,k;
	while(scanf("%d%d",&n,&k) != EOF) {
		int val,le,ri;
		end = -1;
		Node* node;
		int sum = 0;
		for (int i = 0; i < n; i++) {
			scanf("%d%d%d",&val,&le,&ri);
			node = new Node;
			node->index = i + 1;
			node->val = val;
			node->left = le;
			node->right = ri;
			a[i + 1] = node;
		}
		ChangeTree(a[1]);
		printf("result:\n");
		DFS(a[1],k,sum);
	}
	return 0;
}
